### Evening (2-3 hours): Basic Syntax

```
#include <stdio.h>

int main() {
    int number = 42;
    char letter = 'A';
    float pi = 3.14;

    printf("Number: %d\n", number);
    printf("Letter: %c\n", letter);
    printf("Pi: %.2f\n", pi);

    return 0;
}

```
